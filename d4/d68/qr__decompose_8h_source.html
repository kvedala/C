102 /**
103  * Decompose matrix \f$A\f$ using [Gram-Schmidt
104  *process](https://en.wikipedia.org/wiki/QR_decomposition).
105  *
106  * \f{eqnarray*}{
107  * \text{given that}\quad A &=&
108  *\left[\mathbf{a}_1,\mathbf{a}_2,\ldots,\mathbf{a}_{N-1},\right]\\
109  * \text{where}\quad\mathbf{a}_i &=&
110  *\left[a_{0i},a_{1i},a_{2i},\ldots,a_{(M-1)i}\right]^T\quad\ldots\mbox{(column
111  *vectors)}\\
112  * \text{then}\quad\mathbf{u}_i &=& \mathbf{a}_i
113  *-\sum_{j=0}^{i-1}\text{proj}_{\mathbf{u}_j}\mathbf{a}_i\\
114  * \mathbf{e}_i &=&\frac{\mathbf{u}_i}{\left|\mathbf{u}_i\right|}\\
115  * Q &=& \begin{bmatrix}\mathbf{e}_0 & \mathbf{e}_1 & \mathbf{e}_2 & \dots &
116  *\mathbf{e}_{N-1}\end{bmatrix}\\
117  * R &=& \begin{bmatrix}\langle\mathbf{e}_0\,,\mathbf{a}_0\rangle &
118  *\langle\mathbf{e}_1\,,\mathbf{a}_1\rangle &
119  *\langle\mathbf{e}_2\,,\mathbf{a}_2\rangle & \dots \\
120  * 0 & \langle\mathbf{e}_1\,,\mathbf{a}_1\rangle &
121  *\langle\mathbf{e}_2\,,\mathbf{a}_2\rangle & \dots\\
122  * 0 & 0 & \langle\mathbf{e}_2\,,\mathbf{a}_2\rangle & \dots\\
123  * \vdots & \vdots & \vdots & \ddots
124  * \end{bmatrix}\\
125  * \f}
126  **/
